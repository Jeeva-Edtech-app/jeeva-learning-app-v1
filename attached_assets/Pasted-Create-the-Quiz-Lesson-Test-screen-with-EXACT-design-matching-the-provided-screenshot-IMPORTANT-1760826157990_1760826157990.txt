```typescript
import React, { useState, useEffect } from 'react';
import { View, Text, TouchableOpacity, ScrollView, Alert } from 'react-native';
import { Ionicons } from '@expo/vector-icons';
import { useRoute, useNavigation } from '@react-navigation/native';

export default function QuizScreen() {
  const route = useRoute();
  const navigation = useNavigation();
  const { topicId, questions } = route.params;
  
  const [currentIndex, setCurrentIndex] = useState(0);
  const [selectedAnswerId, setSelectedAnswerId] = useState(null);
  const [isSubmitted, setIsSubmitted] = useState(false);
  const [timeRemaining, setTimeRemaining] = useState(60);
  const [results, setResults] = useState([]);
  
  const currentQuestion = questions[currentIndex];
  
  // Timer logic
  useEffect(() => {
    if (isSubmitted) return;
    
    const interval = setInterval(() => {
      setTimeRemaining((prev) => {
        if (prev <= 1) {
          handleAutoSubmit();
          return 60;
        }
        return prev - 1;
      });
    }, 1000);
    
    return () => clearInterval(interval);
  }, [currentIndex, isSubmitted]);
  
  const handleAnswerSelect = (optionId) => {
    if (!isSubmitted) {
      setSelectedAnswerId(optionId);
    }
  };
  
  const handleNext = () => {
    if (!selectedAnswerId && !isSubmitted) {
      Alert.alert('Please select an answer');
      return;
    }
    
    if (!isSubmitted) {
      // Submit answer
      setIsSubmitted(true);
      const isCorrect = selectedAnswerId === currentQuestion.correct_answer_id;
      setResults([...results, { questionId: currentQuestion.id, isCorrect }]);
    } else {
      // Move to next question
      if (currentIndex < questions.length - 1) {
        setCurrentIndex(currentIndex + 1);
        setSelectedAnswerId(null);
        setIsSubmitted(false);
        setTimeRemaining(60);
      } else {
        // Finish quiz
        navigation.navigate('QuizResults', { results, questions });
      }
    }
  };
  
  const handleAutoSubmit = () => {
    setIsSubmitted(true);
    const isCorrect = selectedAnswerId === currentQuestion.correct_answer_id;
    setResults([...results, { questionId: currentQuestion.id, isCorrect }]);
  };
  
  return (
    <View style={styles.container}>
      {/* Header */}
      <View style={styles.header}>
        <TouchableOpacity onPress={() => navigation.goBack()}>
          <Ionicons name="arrow-back" size={24} color="#1F2937" />
        </TouchableOpacity>
        <Text style={styles.headerTitle}>Lesson Test</Text>
      </View>
      
      {/* Question Header */}
      <View style={styles.questionHeader}>
        <Text style={styles.questionNumber}>{currentIndex + 1}</Text>
        <View style={styles.timerContainer}>
          <Ionicons name="timer-outline" size={18} color="#EF4444" />
          <Text style={styles.timerText}>
            {Math.floor(timeRemaining / 60).toString().padStart(2, '0')}:
            {(timeRemaining % 60).toString().padStart(2, '0')}
          </Text>
        </View>
      </View>
      
      <ScrollView style={styles.content}>
        {/* Question Text */}
        <View style={styles.questionSection}>
          <Text style={styles.questionText}>{currentQuestion.question_text}</Text>
        </View>
        
        {/* Answer Options */}
        <View style={styles.optionsContainer}>
          {currentQuestion.question_options.map((option) => (
            <AnswerOption
              key={option.id}
              option={option}
              isSelected={selectedAnswerId === option.id}
              isSubmitted={isSubmitted}
              isCorrect={option.is_correct}
              onPress={() => handleAnswerSelect(option.id)}
            />
          ))}
        </View>
        
        {/* Explanation (after submission) */}
        {isSubmitted && (
          <View style={styles.explanationCard}>
            <Text style={styles.explanationLabel}>Explanation:</Text>
            <Text style={styles.explanationText}>
              {currentQuestion.explanation}
            </Text>
          </View>
        )}
      </ScrollView>
      
      {/* Navigation Buttons */}
      <View style={styles.buttonContainer}>
        {currentIndex > 0 && (
          <TouchableOpacity style={styles.backButton} onPress={() => setCurrentIndex(currentIndex - 1)}>
            <Text style={styles.buttonText}>Back</Text>
          </TouchableOpacity>
        )}
        <TouchableOpacity 
          style={[styles.nextButton, !selectedAnswerId && !isSubmitted && styles.disabledButton]}
          onPress={handleNext}
        >
          <Text style={styles.buttonText}>
            {currentIndex === questions.length - 1 ? 'Finish' : 'Next'}
          </Text>
        </TouchableOpacity>
      </View>
    </View>
  );
}

const AnswerOption = ({ option, isSelected, isSubmitted, isCorrect, onPress }) => {
  const getOptionStyle = () => {
    if (isSubmitted) {
      if (isCorrect) {
        return styles.correctOption;
      } else if (isSelected && !isCorrect) {
        return styles.incorrectOption;
      }
    } else if (isSelected) {
      return styles.selectedOption;
    }
    return styles.defaultOption;
  };
  
  return (
    <TouchableOpacity 
      style={[styles.optionCard, getOptionStyle()]}
      onPress={onPress}
      disabled={isSubmitted}
    >
      <Text style={styles.optionText}>{option.option_text}</Text>
      {isSubmitted && isCorrect && (
        <Ionicons name="checkmark-circle" size={24} color="#10B981" />
      )}
      {isSubmitted && isSelected && !isCorrect && (
        <Ionicons name="close-circle" size={24} color="#EF4444" />
      )}
    </TouchableOpacity>
  );
};

const styles = StyleSheet.create({
  questionText: {
    fontFamily: 'Inter-Regular',
    fontSize: 18,
    color: '#1F2937',
    lineHeight: 28.8, // 1.6 * 18
  },
  optionText: {
    fontFamily: 'Inter-Regular',
    fontSize: 15,
    color: '#6B7280',
  },
  // ... add all other styles
});

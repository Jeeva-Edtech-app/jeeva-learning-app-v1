import { ScrollView, View, Text, TouchableOpacity } from 'react-native';
import { Ionicons, MaterialCommunityIcons } from '@expo/vector-icons';
import { useSubscription } from '@/hooks/useSubscription';

const PRACTICE_MODULES = [
  {
    id: 'numeracy',
    title: 'Numeracy',
    description: 'Timed practice with dosage, IV flow rates and unit conversions',
    icon: 'calculator',
    bgColor: '#D1FAE5',
    iconBgColor: '#10B981',
    isPremium: false, // Free in trial
  },
  {
    id: 'clinical',
    title: 'Clinical',
    description: 'UK clinical scenarios, patient care, and nursing protocols',
    icon: 'stethoscope',
    bgColor: '#FED7AA',
    iconBgColor: '#EF4444',
    isPremium: true,
  },
  {
    id: 'professional',
    title: 'Professional Standards',
    description: 'NMC Code principles, ethics, and professional conduct',
    icon: 'shield-check',
    bgColor: '#EDE9FE',
    iconBgColor: '#7C3AED',
    isPremium: true,
  },
];

export default function PracticeModuleScreen({ navigation }) {
  const { isTrialUser } = useSubscription();

  const handleModuleTap = (module) => {
    if (isTrialUser && module.isPremium) {
      // Show upgrade prompt
      navigation.navigate('UpgradePrompt');
    } else {
      navigation.navigate('TopicsList', { moduleId: module.id });
    }
  };

  return (
    <ScrollView style={styles.container}>
      {/* Header */}
      <View style={styles.header}>
        <TouchableOpacity onPress={() => navigation.goBack()}>
          <Ionicons name="arrow-back" size={24} color="#1F2937" />
        </TouchableOpacity>
        <Text style={styles.headerTitle}>Practice Module</Text>
      </View>

      {/* Hero Card */}
      <View style={styles.heroCard}>
        <Text style={styles.heroTitle}>PRACTICE MODE</Text>
        <Text style={styles.heroSubtitle}>
          Practice topic-wise questions at your own pace.
        </Text>
      </View>

      {/* Section Header */}
      <Text style={styles.sectionTitle}>Choose Your Practice Module</Text>
      <Text style={styles.sectionSubtitle}>
        Select the section you want to focus on and start practicing immediately
      </Text>

      {/* Module Cards */}
      <View style={styles.moduleList}>
        {PRACTICE_MODULES.map((module) => (
          <ModuleCard
            key={module.id}
            module={module}
            onPress={() => handleModuleTap(module)}
            isLocked={isTrialUser && module.isPremium}
          />
        ))}
      </View>
    </ScrollView>
  );
}

const ModuleCard = ({ module, onPress, isLocked }) => (
  <TouchableOpacity 
    style={[styles.moduleCard, { backgroundColor: module.bgColor }]}
    onPress={onPress}
  >
    {/* Icon */}
    <View style={[styles.iconCircle, { backgroundColor: module.iconBgColor }]}>
      <MaterialCommunityIcons name={module.icon} size={28} color="white" />
    </View>

    {/* Content */}
    <View style={styles.cardContent}>
      <Text style={styles.cardTitle}>{module.title}</Text>
      <Text style={styles.cardDescription}>{module.description}</Text>
    </View>

    {/* Chevron or Lock */}
    {isLocked ? (
      <View style={styles.lockIcon}>
        <Ionicons name="lock-closed" size={20} color="#6B7280" />
      </View>
    ) : (
      <View style={styles.chevronCircle}>
        <Ionicons name="chevron-forward" size={24} color="#9CA3AF" />
      </View>
    )}
  </TouchableOpacity>
);

const styles = StyleSheet.create({
  container: {
    flex: 1,
    backgroundColor: '#F9FAFB',
  },
  heroTitle: {
    fontFamily: 'Inter-Bold',
    fontSize: 22,
    color: 'white',
  },
  cardTitle: {
    fontFamily: 'Inter-Bold',
    fontSize: 20,
    color: '#1F2937',
  },
  // ... add all other styles with Inter font family
});
